```typescript
/**
* Parse duration string to milliseconds
* @param duration - Format: "1h", "24h", "7d", "30m", "60s"
* @throws Error if format is invalid
*/
export function parseDuration(duration: string): number {
 const match = duration.match(/^(\d+)(h|d|m|s)$/);
 if (!match) {
   throw new Error(
     `Invalid duration format: ${duration}. Use format like "1h", "24h", "7d"`
   );
 }


 const [, amount, unit] = match;
 const num = parseInt(amount, 10);


 switch (unit) {
   case "s": return num * 1000;
   case "m": return num * 60 * 1000;
   case "h": return num * 60 * 60 * 1000;
   case "d": return num * 24 * 60 * 60 * 1000;
   default: throw new Error(`Unknown duration unit: ${unit}`);
 }
}


/**
* Format milliseconds to human-readable duration
*/
export function formatDuration(ms: number): string {
 if (ms < 1000) return `${ms}ms`;
 if (ms < 60000) return `${(ms / 1000).toFixed(2)}s`;
 if (ms < 3600000) return `${(ms / 60000).toFixed(1)}m`;
 return `${(ms / 3600000).toFixed(1)}h`;
}
```

```typescript
import { homedir } from "node:os";
import { join } from "node:path";
import { readdir, stat } from "node:fs/promises";


export interface TraceDirectoryOptions {
 dir?: string;
}


/**
* Manages trace directory resolution and file listing
*/
export class TraceDirectory {
 private resolvedDir: string;


 constructor(options: TraceDirectoryOptions = {}) {
   this.resolvedDir = this.resolveDirectory(options.dir);
 }


 /**
  * Resolve trace directory
  * Priority: explicit > env > default
  */
 private resolveDirectory(explicitDir?: string): string {
   if (explicitDir) return explicitDir;
  
   const envDir = process.env.AGENT_INSPECT_TRACE_DIR;
   if (envDir) return envDir;
  
   return join(homedir(), ".agent-inspect", "runs");
 }


 /**
  * List all .jsonl files in directory
  * Returns empty array if directory doesn't exist
  */
 async list(): Promise<string[]> {
   try {
     const files = await readdir(this.resolvedDir);
     return files.filter(f => f.endsWith(".jsonl"));
   } catch (err) {
     if ((err as NodeJS.ErrnoException).code === "ENOENT") {
       return [];
     }
     throw err;
   }
 }


 /**
  * Get full path for a file or directory
  */
 getPath(filename?: string): string {
   return filename ? join(this.resolvedDir, filename) : this.resolvedDir;
 }


 /**
  * Get file stats
  */
 async getFileStats(filename: string) {
   const path = this.getPath(filename);
   return await stat(path);
 }
}
```

#### Step 2.1: Trace Filter


**File**: `packages/core/src/trace-filter.ts`


```typescript
import type { TraceMetadata, TraceMetadataStatus } from "./types";
import { parseDuration } from "./utils/duration";


export interface TraceFilterCriteria {
 status?: TraceMetadataStatus;
 name?: string;
 since?: string; // "1h", "24h", "7d"
 before?: string;
 minDuration?: number;
 maxDuration?: number;
 limit?: number;
}


export class TraceFilter {
 /**
  * Check if trace matches criteria
  */
 static matches(metadata: TraceMetadata, criteria: TraceFilterCriteria): boolean {
   // Status filter
   if (criteria.status && metadata.status !== criteria.status) {
     return false;
   }


   // Name filter (substring match, case-insensitive)
   if (criteria.name) {
     const name = metadata.name?.toLowerCase() || "";
     if (!name.includes(criteria.name.toLowerCase())) {
       return false;
     }
   }


   // Time range filters
   if (criteria.since) {
     const sinceMs = parseDuration(criteria.since);
     const cutoff = Date.now() - sinceMs;
     if (!metadata.startedAt || metadata.startedAt < cutoff) {
       return false;
     }
   }


   if (criteria.before) {
     const beforeMs = parseDuration(criteria.before);
     const cutoff = Date.now() - beforeMs;
     if (!metadata.startedAt || metadata.startedAt > cutoff) {
       return false;
     }
   }


   // Duration filters
   if (criteria.minDuration && (!metadata.durationMs || metadata.durationMs < criteria.minDuration)) {
     return false;
   }


   if (criteria.maxDuration && (!metadata.durationMs || metadata.durationMs > criteria.maxDuration)) {
     return false;
   }


   return true;
 }


 /**
  * Filter and sort traces
  * Returns most recent first
  */
 static filter(traces: TraceMetadata[], criteria: TraceFilterCriteria): TraceMetadata[] {
   let filtered = traces.filter(t => this.matches(t, criteria));


   // Sort by start time, most recent first
   filtered.sort((a, b) => (b.startedAt || 0) - (a.startedAt || 0));


   // Apply limit
   if (criteria.limit) {
     filtered = filtered.slice(0, criteria.limit);
   }


   return filtered;
 }
}
```

```typescript
import chalk from "chalk";
import type { TraceMetadata } from "agent-inspect-core";


export interface TableOptions {
 verbose?: boolean;
}


/**
* Internal table renderer using only chalk
* NO cli-table3 dependency
*/
export function renderTable(
 traces: TraceMetadata[],
 options: TableOptions = {}
): string {
 const lines: string[] = [];


 // Header
 const headers = options.verbose
   ? ["Run ID", "Name", "Status", "Duration", "Events", "Started"]
   : ["Run ID", "Name", "Status", "Duration"];


 lines.push(chalk.cyan(formatRow(headers, options.verbose)));
 lines.push(chalk.gray("─".repeat(80)));


 // Rows
 for (const trace of traces) {
   const row = [
     trace.runId.slice(0, 12),
     truncate(trace.name || "-", 20),
     formatStatus(trace.status),
     trace.durationMs ? formatDuration(trace.durationMs) : "-",
   ];


   if (options.verbose) {
     row.push(
       trace.eventCount.toString(),
       trace.startedAt ? new Date(trace.startedAt).toLocaleString() : "-"
     );
   }


   lines.push(formatRow(row, options.verbose));
 }


 return lines.join("\n");
}


function formatRow(cells: string[], verbose: boolean): string {
 const widths = verbose
   ? [14, 22, 12, 10, 8, 20]
   : [14, 22, 12, 10];


 return cells
   .map((cell, i) => padEnd(cell, widths[i]))
   .join(" ");
}


function formatStatus(status: string): string {
 switch (status) {
   case "success": return chalk.green("✔ success");
   case "error": return chalk.red("✖ error");
   case "running": return chalk.yellow("⏳ running");
   case "unknown": return chalk.gray("? unknown");
   default: return status;
 }
}


function truncate(str: string, maxLen: number): string {
 if (str.length <= maxLen) return str;
 return str.slice(0, maxLen - 1) + "…";
}


function padEnd(str: string, width: number): string {
 // Strip ANSI codes for length calculation
 const visibleLength = str.replace(/\x1b\[[0-9;]*m/g, "").length;
 const padding = Math.max(0, width - visibleLength);
 return str + " ".repeat(padding);
}


function formatDuration(ms: number): string {
 if (ms < 1000) return `${ms}ms`;
 if (ms < 60000) return `${(ms / 1000).toFixed(2)}s`;
 return `${(ms / 60000).toFixed(1)}m`;
}
```

```typescript
import { createReadStream } from "node:fs";
import { createInterface } from "node:readline";


/**
* Verify a file is an AgentInspect trace
* Prevents accidental deletion of arbitrary .jsonl files
*/
export async function isAgentInspectTrace(filePath: string): Promise<boolean> {
 try {
   const stream = createInterface({
     input: createReadStream(filePath),
     crlfDelay: Infinity,
   });


   // Read first valid JSON line
   for await (const line of stream) {
     if (!line.trim()) continue;


     try {
       const event = JSON.parse(line);


       // Check for AgentInspect markers
       // Option 1: Schema version
       if (event.schemaVersion) return true;


       // Option 2: Known event patterns
       // IMPORTANT: Use ACTUAL event names from your schema
       if (event.event?.startsWith("run_") || event.event?.startsWith("step_")) {
         return true;
       }


       // Option 3: Known structure
       if (event.runId || event.stepId) {
         return true;
       }


       // First line parsed but doesn't look like AgentInspect
       return false;
     } catch {
       continue;
     }
   }


   // No valid JSON found
   return false;
 } catch {
   return false;
 }
}
```

#### Step 4.2: Clean Command


**File**: `packages/cli/src/commands/clean.ts`


```typescript
import { Command } from "commander";
import { createInterface } from "node:readline/promises";
import { stdin as input, stdout as output } from "node:process";
import { unlink } from "node:fs/promises";
import { TraceDirectory, TraceFilter, extractMetadata, isAgentInspectTrace } from "agent-inspect-core";
import { parseDuration } from "agent-inspect-core/utils";
import { renderTable } from "../renderers/table-renderer";


export function createCleanCommand() {
 return new Command("clean")
   .description("Remove old trace files")
   .option("--dir <path>", "Trace directory")
   .option("--older-than <duration>", "Remove runs older than duration (e.g., 7d, 30d)")
   .option("--keep <number>", "Keep this many most recent runs")
   .option("--dry-run", "Show what would be deleted without deleting")
   .action(async (options) => {
     try {
       await handleClean(options);
     } catch (err) {
       console.error("Error cleaning runs:", (err as Error).message);
       process.exit(1);
     }
   });
}


async function handleClean(options: any) {
 if (!options.olderThan && !options.keep) {
   console.error("Error: Must specify either --older-than or --keep");
   process.exit(1);
 }


 const traceDir = new TraceDirectory({ dir: options.dir });
 const files = await traceDir.list();


 if (files.length === 0) {
   console.log("No runs to clean.");
   return;
 }


 // Get metadata for all files
 const metadataPromises = files.map(file =>
   extractMetadata(traceDir.getPath(file), false)
 );
 const allMetadata = await Promise.all(metadataPromises);


 // Determine what to delete
 let toDelete: typeof allMetadata = [];


 if (options.olderThan) {
   const cutoffMs = Date.now() - parseDuration(options.olderThan);
   toDelete = allMetadata.filter(m => m.startedAt && m.startedAt < cutoffMs);
 }


 if (options.keep) {
   const keepCount = parseInt(options.keep, 10);
   const sorted = [...allMetadata].sort((a, b) => (b.startedAt || 0) - (a.startedAt || 0));
   toDelete = sorted.slice(keepCount);
 }


 if (toDelete.length === 0) {
   console.log("No runs match deletion criteria.");
   return;
 }


 // Verify all files are AgentInspect traces
 console.log("Verifying files are AgentInspect traces...");
 const verifiedToDelete = [];
 for (const metadata of toDelete) {
   if (await isAgentInspectTrace(metadata.filePath)) {
     verifiedToDelete.push(metadata);
   } else {
     console.warn(`Skipping ${metadata.runId}: not an AgentInspect trace`);
   }
 }


 if (verifiedToDelete.length === 0) {
   console.log("No verified AgentInspect traces to delete.");
   return;
 }


 // Show what will be deleted
 console.log(`\nFound ${verifiedToDelete.length} runs to delete:\n`);
 console.log(renderTable(verifiedToDelete, { verbose: false }));


 if (options.dryRun) {
   console.log("\n--dry-run: No files were deleted.");
   return;
 }


 // Confirm deletion
 const rl = createInterface({ input, output });
 const answer = await rl.question("\nDelete these files? (y/N) ");
 rl.close();


 if (answer.toLowerCase() !== "y") {
   console.log("Cancelled.");
   return;
 }


 // Delete files
 for (const metadata of verifiedToDelete) {
   await unlink(metadata.filePath);
 }


 console.log(`\nDeleted ${verifiedToDelete.length} files.`);
}
```
