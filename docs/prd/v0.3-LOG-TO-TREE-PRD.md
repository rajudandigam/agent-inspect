```typescript
// CORRECTED: JSON-first, simple structure
import { readFileSync } from "node:fs";


interface LogLine {
 timestamp: number;
 event: string;
 decisionId?: string;
 durationMs?: number;
 [key: string]: unknown;
}


function parseJsonLog(filePath: string): LogLine[] {
 const content = readFileSync(filePath, "utf-8");
 return content
   .split("\n")
   .filter(line => line.trim())
   .map(line => JSON.parse(line));
}


function groupByRunId(logs: LogLine[]): Map<string, LogLine[]> {
 const groups = new Map<string, LogLine[]>();
  for (const log of logs) {
   const runId = log.decisionId || log.requestId || "unknown";
   if (!groups.has(runId)) {
     groups.set(runId, []);
   }
   groups.get(runId)!.push(log);
 }
  return groups;
}


function renderTimeline(logs: LogLine[]): string {
 const lines: string[] = [];
 const runId = logs[0]?.decisionId || "unknown";
  lines.push(`Run decision=${runId.slice(0, 8)}`);
  for (let i = 0; i < logs.length; i++) {
   const log = logs[i];
   const isLast = i === logs.length - 1;
   const prefix = isLast ? "└─" : "├─";
  
   // Extract event name
   const eventParts = log.event.split(".");
   const kind = eventParts[1];  // tool, llm, agent
   const name = eventParts.slice(2).join(":");
  
   let line = `${prefix} ${kind}:${name}`;
  
   // Add metadata
   if (log.messageCount) line += ` msgs=${log.messageCount}`;
   if (log.durationMs) line += ` ✔ ${(log.durationMs / 1000).toFixed(2)}s`;
  
   lines.push(line);
   lines.push(`   confidence: correlated (same decisionId)`);
 }
  return lines.join("\n");
}


// Run prototype
const logs = parseJsonLog("sample-json.log");
const groups = groupByRunId(logs);


for (const [runId, runLogs] of groups) {
 console.log(renderTimeline(runLogs));
 console.log();
}
```

```typescript
import { createReadStream } from "node:fs";
import { createInterface } from "node:readline";


export interface RawLogEvent {
 raw: string;
 line: number;
 parsed: Record<string, unknown>;
}


export interface ParserOptions {
 quiet?: boolean;
 warnings?: "none" | "summary" | "all";  // CORRECTED: controllable warnings
}


/**
* JSON log parser (FIRST-CLASS support)
* Expects line-delimited JSON
*/
export class JsonLogParser {
 private skippedLines: number = 0;
 private options: ParserOptions;


 constructor(options: ParserOptions = {}) {
   this.options = {
     quiet: false,
     warnings: "summary",
     ...options,
   };
 }


 async *parse(filePath: string): AsyncGenerator<RawLogEvent> {
   const stream = createInterface({
     input: createReadStream(filePath),
     crlfDelay: Infinity,
   });


   let lineNumber = 0;


   for await (const line of stream) {
     lineNumber++;
    
     if (!line.trim()) continue;


     try {
       const parsed = JSON.parse(line);
       yield {
         raw: line,
         line: lineNumber,
         parsed,
       };
     } catch (err) {
       this.skippedLines++;
      
       if (this.options.warnings === "all") {
         console.warn(`Line ${lineNumber}: Invalid JSON - ${line.slice(0, 50)}...`);
       }
      
       continue;
     }
   }


   // Summary warnings
   if (this.options.warnings === "summary" && this.skippedLines > 0) {
     console.warn(`Skipped ${this.skippedLines} unparsable lines. Use --verbose for details.`);
   }
 }


 getSkippedCount(): number {
   return this.skippedLines;
 }
}
```

```typescript
/**
* log4js parser (BEST-EFFORT support)
* Extracts JSON payloads from log4js text format
*
* CORRECTED: Only parses valid JSON, skips JS-object-style strings
*/
export class Log4jsParser {
 private skippedLines: number = 0;
 private options: ParserOptions;


 constructor(options: ParserOptions = {}) {
   this.options = {
     quiet: false,
     warnings: "summary",
     ...options,
   };
 }


 async *parse(filePath: string): AsyncGenerator<RawLogEvent> {
   const stream = createInterface({
     input: createReadStream(filePath),
     crlfDelay: Infinity,
   });


   let lineNumber = 0;


   for await (const line of stream) {
     lineNumber++;
    
     if (!line.trim()) continue;


     // log4js format: "2026-05-05 12:20:18.130 [INFO] [default] - Message {json}"
     const jsonMatch = line.match(/\{.*\}/);
    
     if (!jsonMatch) {
       this.skippedLines++;
       continue;
     }


     try {
       const parsed = JSON.parse(jsonMatch[0]);  // CORRECTED: Only valid JSON
      
       // Extract timestamp from log4js prefix if available
       const timestampMatch = line.match(/^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}\.\d{3})/);
       if (timestampMatch && !parsed.timestamp) {
         parsed.timestamp = new Date(timestampMatch[1]).getTime();
       }


       yield {
         raw: line,
         line: lineNumber,
         parsed,
       };
     } catch (err) {
       this.skippedLines++;
      
       if (this.options.warnings === "all") {
         console.warn(`Line ${lineNumber}: Invalid JSON in log4js line`);
       }
      
       continue;
     }
   }


   if (this.options.warnings === "summary" && this.skippedLines > 0) {
     console.warn(`Skipped ${this.skippedLines} lines (no valid JSON). Use --verbose for details.`);
   }
 }


 getSkippedCount(): number {
   return this.skippedLines;
 }
}
```

```typescript
import { InspectEvent, InspectKind, EventSource } from "../types/inspect-event";
import { LogIngestConfig } from "../types/log-config";
import { RawLogEvent } from "../parsers";
import { nanoid } from "nanoid";  // CORRECTED: use nanoid, not uuid


export class EventNormalizer {
 constructor(
   private config: LogIngestConfig,
   private sourceType: "json-log" | "log4js"  // CORRECTED: preserve source type
 ) {}


 normalize(raw: RawLogEvent): InspectEvent | null {
   // Extract runId
   const runId = this.extractRunId(raw.parsed);
   if (!runId) {
     return null;  // Skip events without runId
   }


   // Extract event name
   const eventName = this.extractEventName(raw.parsed);
   if (!eventName) {
     return null;
   }


   // Find matching mapping
   const mapping = this.findMapping(eventName);


   // Extract timestamp
   const timestamp = this.extractTimestamp(raw.parsed);


   // Build event
   const event: InspectEvent = {
     eventId: nanoid(),  // CORRECTED: use nanoid
     runId,
     name: mapping?.name || this.simplifyEventName(eventName),
     kind: mapping?.kind || "LOG",
     timestamp,
     confidence: "correlated",  // Will be refined by tree builder
     source: {
       type: this.sourceType,  // CORRECTED: preserve actual parser type
       line: raw.line,
     },
     attributes: this.extractAttributes(raw.parsed),
   };


   // Extract optional fields
   if (this.config.parentIdKey) {
     const parentId = raw.parsed[this.config.parentIdKey];
     if (parentId && typeof parentId === "string") {
       event.parentId = parentId;
       event.confidence = "explicit";  // Has explicit parent
     }
   }


   if (this.config.statusKey) {
     const status = raw.parsed[this.config.statusKey];
     if (status === "ok" || status === "success") event.status = "ok";
     else if (status === "error" || status === "failed") event.status = "error";
     else if (status === "running") event.status = "running";
   }


   if (mapping?.status) {
     event.status = mapping.status;
   }


   // CORRECTED: Only set duration if explicit
   if (this.config.durationKey) {
     const duration = raw.parsed[this.config.durationKey];
     if (typeof duration === "number") {
       event.durationMs = duration;
     }
   }


   return event;
 }


 private extractRunId(parsed: Record<string, unknown>): string | null {
   for (const key of this.config.runIdKeys) {
     const value = parsed[key];
     if (value && typeof value === "string") {
       return value;
     }
   }
   return null;
 }


 private extractEventName(parsed: Record<string, unknown>): string | null {
   const value = parsed[this.config.eventKey];
   return value && typeof value === "string" ? value : null;
 }


 private extractTimestamp(parsed: Record<string, unknown>): number {
   const key = this.config.timestampKey || "timestamp";
   const value = parsed[key];
  
   if (typeof value === "number") return value;
   if (typeof value === "string") return new Date(value).getTime();
  
   return Date.now();
 }


 private findMapping(eventName: string) {
   if (!this.config.mappings) return null;


   // Exact match first
   if (this.config.mappings[eventName]) {
     return this.config.mappings[eventName];
   }


   // Wildcard match
   for (const [pattern, mapping] of Object.entries(this.config.mappings)) {
     if (this.matchesPattern(eventName, pattern)) {
       return mapping;
     }
   }


   return null;
 }


 private matchesPattern(eventName: string, pattern: string): boolean {
   const regex = new RegExp(
     "^" + pattern.replace(/\./g, "\\.").replace(/\*/g, ".*") + "$"
   );
   return regex.test(eventName);
 }


 private simplifyEventName(eventName: string): string {
   const parts = eventName.split(".");
   return parts[parts.length - 1] || eventName;
 }


 private extractAttributes(parsed: Record<string, unknown>): Record<string, unknown> {
   const skip = new Set([
     ...this.config.runIdKeys,
     this.config.eventKey,
     this.config.timestampKey,
     this.config.messageKey,
     this.config.levelKey,
     this.config.parentIdKey,
     this.config.durationKey,
     this.config.statusKey,
   ]);


   const attributes: Record<string, unknown> = {};
   for (const [key, value] of Object.entries(parsed)) {
     if (!skip.has(key)) {
       attributes[key] = value;
     }
   }


   return attributes;
 }
}
```

### Phase 4: Tree Building (Week 4) - CORRECTED: Conservative


#### Step 4.1: Tree Builder (FLAT BY DEFAULT)


**File**: `packages/core/src/tree/tree-builder.ts`


```typescript
import { InspectEvent, InspectNode, InspectRunTree } from "../types/inspect-event";


/**
* CORRECTED: Conservative tree builder
* Default: flat timeline grouped by run
* Only nests when explicit parentId or config mapping
*/
export class TreeBuilder {
 constructor(private heuristicWindowMs: number = 2000) {}


 build(events: InspectEvent[]): InspectRunTree[] {
   // Group by runId
   const byRun = new Map<string, InspectEvent[]>();
   for (const event of events) {
     if (!byRun.has(event.runId)) {
       byRun.set(event.runId, []);
     }
     byRun.get(event.runId)!.push(event);
   }


   // Build tree for each run
   const trees: InspectRunTree[] = [];
   for (const [runId, runEvents] of byRun) {
     trees.push(this.buildRunTree(runId, runEvents));
   }


   return trees;
 }


 private buildRunTree(runId: string, events: InspectEvent[]): InspectRunTree {
   // Sort by timestamp
   events.sort((a, b) => a.timestamp - b.timestamp);


   // Find or create root
   let root = events.find(e => e.kind === "RUN");
   if (!root) {
     root = {
       eventId: `${runId}_root`,
       runId,
       name: runId.slice(0, 12),
       kind: "RUN",
       timestamp: events[0]?.timestamp || Date.now(),
       confidence: "explicit",
       source: { type: "manual" },
     };
   }


   // Build node map
   const nodes = new Map<string, InspectNode>();
   const rootNode: InspectNode = {
     event: root,
     children: [],
     depth: 0,
   };
   nodes.set(root.eventId, rootNode);


   // CORRECTED: Flat timeline by default
   for (const event of events) {
     if (event === root) continue;


     const node: InspectNode = {
       event,
       children: [],
       depth: 1,  // Default: direct child of root
     };


     // Only nest if explicit parent
     if (event.parentId) {
       const parent = nodes.get(event.parentId);
       if (parent) {
         parent.children.push(node);
         node.depth = parent.depth + 1;
       } else {
         // Parent not found, attach to root
         rootNode.children.push(node);
       }
     } else {
       // No explicit parent, attach to root (flat timeline)
       rootNode.children.push(node);
     }


     nodes.set(event.eventId, node);
   }


   // Calculate metadata
   const confidenceBreakdown: Record<string, number> = {
     explicit: 0,
     correlated: 0,
     heuristic: 0,
     unknown: 0,
   };


   const kinds: Record<string, number> = {};


   for (const event of events) {
     confidenceBreakdown[event.confidence]++;
     kinds[event.kind] = (kinds[event.kind] || 0) + 1;
   }


   return {
     runId,
     name: root.name,
     status: this.determineRunStatus(events),
     startedAt: events[0]?.timestamp,
     endedAt: events[events.length - 1]?.timestamp,
     durationMs: events[events.length - 1]?.timestamp - events[0]?.timestamp,
     children: rootNode.children,
     metadata: {
       totalEvents: events.length,
       confidenceBreakdown,
       kinds,
     },
   };
 }


 private determineRunStatus(events: InspectEvent[]): "running" | "ok" | "error" {
   const hasError = events.some(e => e.status === "error" || e.kind === "ERROR");
   if (hasError) return "error";


   const hasEnd = events.some(e => e.kind === "RESULT" || e.name?.includes("ended"));
   return hasEnd ? "ok" : "running";
 }
}
```

```typescript
* CORRECTED: Named strategies only, no function strings
*/
export class Redactor {
 constructor(private rules: RedactionRule[] = []) {}


 redact(obj: Record<string, unknown>): Record<string, unknown> {
   const result: Record<string, unknown> = {};


   for (const [key, value] of Object.entries(obj)) {
     const rule = this.findRule(key);
    
     if (!rule) {
       result[key] = value;
       continue;
     }


     if (typeof rule === "string") {
       // Simple string rule: full redaction
       result[key] = "[REDACTED]";
     } else {
       // Named strategy
       result[key] = this.applyStrategy(value, rule.strategy, rule.keep);
     }
   }


   return result;
 }


 private findRule(key: string): RedactionRule | null {
   for (const rule of this.rules) {
     if (typeof rule === "string") {
       if (key === rule) return rule;
     } else {
       if (key === rule.key) return rule;
     }
   }
   return null;
 }


 private applyStrategy(
   value: unknown,
   strategy: RedactionStrategy,
   keep?: number
 ): string {
   const str = String(value);


   switch (strategy) {
     case "full":
       return "[REDACTED]";
    
     case "prefix":
       const keepChars = keep || 8;
       if (str.length <= keepChars) return str;
       return str.slice(0, keepChars) + "…";
    
     case "hash":
       return createHash("sha256").update(str).digest("hex").slice(0, 8);
    
     default:
       return "[REDACTED]";
   }
 }
}
```

```typescript
async function handleLogs(file: string, options: any) {
 // Load config
 const config = options.config
   ? await LogConfigLoader.load(options.config)
   : LogConfigLoader.createDefault(
       options.runIdKey ? [options.runIdKey] : ["decisionId", "requestId"],
       options.eventKey || "event"
     );


 // Create parser
 const parser = LogParserFactory.create(options.format, file);
 parser.options.warnings = options.warnings;


 // Parse events
 const events = [];
 for await (const raw of parser.parse(file)) {
   const normalizer = new EventNormalizer(config, parser.sourceType);
   const event = normalizer.normalize(raw);
   if (event) events.push(event);
 }


 if (events.length === 0) {
   console.log("No events parsed.");
   return;
 }


 // Redact
 const redactor = new Redactor(config.redact);
 for (const event of events) {
   event.attributes = redactor.redact(event.attributes || {});
 }


 // Build tree
 const builder = new TreeBuilder(config.heuristicWindowMs);
 const trees = builder.build(events);


 // Render
 if (options.json) {
   console.log(JSON.stringify(trees, null, 2));
 } else if (options.summary) {
   for (const tree of trees) {
     console.log(renderSummary(tree));
   }
 } else {
   for (const tree of trees) {
     console.log(renderTree(tree));
   }
 }
}
```
